```javascript
const getBalance = () => {
  const url = getNodeRpcURL();
  const connection = new Connection(url);

  const publicKey = new PublicKey(value);

  connection.getBalance(publicKey)
    .then((balance) => {
      setBalance(balance / DECIMAL_OFFSET);
    })
    .catch((error) => {
      console.log(error);
      setBalance(null);
    });
}
```
